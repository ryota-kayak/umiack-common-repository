uniform sampler2D uTexture; uniform vec2 uResolution; uniform vec2 uDrop; // Drop position (normalized [0,1]) uniform vec2 uLight; // Light position (normalized [0,1]) void main() { vec2 uv = gl_FragCoord.xy / uResolution.xy; vec2 dropCenter = uDrop; float r = 0.20; // Drop radius (normalized) float d = distance(uv, dropCenter); float ratio = d / r; // <-- Moved here: always defined vec2 uv2 = uv; // Lens/magnifier effect if (d < r) { float lensStrength = 0.18; uv2 = dropCenter + (uv - dropCenter) * (1.0 - lensStrength * (1.0 - ratio*ratio)); } vec3 col = texture2D(uTexture, uv2).rgb; // Lighting float shading = 1.0; float highlight = 0.0; if (d < r) { // Surface normal of drop (approximate as hemisphere facing +z) vec2 n = normalize(uv - dropCenter); vec3 normal = normalize(vec3(n, sqrt(max(0.0, 1.0 - dot(n, n))))); // Light direction (from drop center) vec2 lightVec2D = normalize(uLight - dropCenter); vec3 lightDir = normalize(vec3(lightVec2D, 0.6)); // Light slightly in front // Simple diffuse shading (optional) shading = 0.95 + 0.35 * max(0.0, dot(normal, lightDir)); // Phong-like specular vec3 viewDir = vec3(0.0,0.0,1.0); vec3 halfDir = normalize(lightDir + viewDir); float spec = pow(max(0.0, dot(normal, halfDir)), 40.0); highlight = spec * (1.0 - ratio) * 2.0; // Sharper at drop center } // Add highlight and soft shading col = col * shading + vec3(highlight); // Add caustic rim if (abs(d - r) < 0.012) { float caustic = 0.22 * (1.0 - smoothstep(0.0, 0.012, abs(d - r))); col += vec3(caustic, caustic, caustic*1.1); } gl_FragColor = vec4(col, 1.0); } WebGL Water Drop with Point Light (Fixed)
